289 if (constraints.has_max_length && len > constraints.max_length)
290 {
291 error = "String length " + std::to_string(len) + " is above maximum " +
292 std::to_string(constraints.max_length);
293 return false;
294 }
295
296 if (constraints.has_allowed)
297 {
298 bool found = false;
299 for (const auto& allowed : constraints.allowed_values)
300 {
301 if (value == allowed)

303 found = true;
304 break;
305 }
306 }
307
308 if (!found)
309 {
310 error = "Value '" + value + "' is not in allowed set";
311 return false;
